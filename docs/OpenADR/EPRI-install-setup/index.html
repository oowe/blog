Download EPRI VTN github#

Install Java7#

  1. wget https://files-cdn.liferay.com/mirrors/download.oracle.com/otn-pub/java/jdk/7u80-b15/jdk-7u80-linux-x64.tar.gz
  2. tar zxf jdk-7u80-linux-x64.tar.gz
  3. sudo mkdir /usr/lib/jvm
  4. sudo mv jdk1.7.0_80/ /usr/lib/jvm
  5. sudo mv /usr/lib/jvm/jdk1.7.0_80/ /usr/lib/jvm/java-7-sun
  6. sudo apt install -y vim
  7. vim ~/.bashrc
  8. add text
    export JAVA_HOME=/usr/lib/jvm/java-7-sunexport JRE_HOME=${JAVA_HOME}/jreexport CLASSPATH=.:${JAVA_HOME}/lib:${JRE_HOME}/libexport PATH=${JAVA_HOME}/bin:$PATH
  9. source ~/.bashrc
  10. sudo update-alternatives --install /usr/bin/javac javac /usr/lib/jvm/java-7-sun/bin/javac 300
  11. sudo update-alternatives --install /usr/bin/jar jar /usr/lib/jvm/java-7-sun/bin/jar 300
  12. sudo update-alternatives --config java
  13. java -version

Install pgAdmin4#

Install rvm#

  1. sudo apt install -y gnupg2
  2. gpg2 --recv-keys 409B6B1796C275462A1703113804BB82D39DC0E3 7D2BAF1CF37B13E2069D6956105BD0E739499BDB
  3. curl -sSL https://get.rvm.io | bash -s stable
  4. echo source ~/.rvm/scripts/rvm >> ~/.bashrc
  5. source ~/.bashrc
  6. rvm version
