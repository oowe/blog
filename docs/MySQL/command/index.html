修改root密碼sudo mysqlSHOW VARIABLES LIKE 'validate_password%';set global validate_password.policy=0;set global validate_password.length=1;ALTER USER 'root'@'localhost' IDENTIFIED BY '{password}';重新載入使用者權限設定FLUSH PRIVILEGES;
#in mysql 5.7set global validate_password_policy=0;set global validate_password._ength=1;


新建帳號CREATE USER '{username}'@'localhost' IDENTIFIED WITH mysql_native_password BY '{password}';
登入mysqlmysql  -h 192.168.11.1 -u {username} -p{password} {dbname}
顯示所有資料庫(DATABASE)show databases;
建立資料庫(DATABASE)CREATE DATABASE `{dbname}`;
更名資料庫RENAME TABLE `old_name` TO `new_name`;
指派DB(EMS)的全部權限給使用者GRANT ALL PRIVILEGES ON EMS.* TO '{username}'@'%';指派全部DB權限GRANT ALL PRIVILEGES ON *.* TO '{username}'@'%';重新載入使用者權限設定FLUSH PRIVILEGES;
登入Container內的方法CREATE USER '{username}'@'%' IDENTIFIED BY '{password}';ALTER USER '{username}'@'%' IDENTIFIED WITH mysql_native_password BY '{password}';GRANT ALL ON *.* TO '{username}'@'%';
(解决Node.js mysql客户端不支持认证协议引发的“ER_NOT_SUPPORTED_AUTH_MODE”问题)
**Reference from https://waylau.com/node.js-mysql-client-does-not-support-authentication-protocol/**

進入DATABASE;use {dbname};
顯示資料表(TABLE);show tables;
建立資料表(TABLE)CREATE TABLE AIR_CONDITIONER_PROJECT ( CITY_NAME char(20) NOT NULL,  SCHOOL_NAME char(20) NOT NULL, CLASSROOM_NUMBER char(20) NOT NULL, AMMETER_UID char(30) NOT NULL, PRIMARY KEY(AMMETER_UID) );
建立資料#標準INSERT INTO AIR_CONDITIONER_PROJECT (CITY_NAME, SCHOOL_NAME, CLASSROOM_NUMBER, AMMETER_UID) VALUES ("宜蘭縣", "羅東國小", "101", "00:00:00:00:00:01");#簡略版並多筆INSERT INTO AIR_CONDITIONER_PROJECT VALUES ("宜蘭縣", "公正國小", "101","00:00:00:00:00:02"), ("新北市", "新店國小", "101", "00:00:00:00:00:03");
INSERT INTO AIR_CONDITIONER_PROJECT VALUES ("新北市", "新店國小", "102","00:00:00:00:00:04"), ("新北市", "網溪國小", "101", "00:00:00:00:00:05"),("澎湖縣", "馬公國小", "101", "00:00:00:00:00:06");
查欄位格式(TABLE)describe AIR_CONDITIONER_PROJECT;desc AIR_CONDITIONER_PROJECT;
查詢資料(TABLE)#全部SELECT * FROM AIR_CONDITIONER_PROJECT;#特定條件select * from AIR_CONDITIONER_PROJECT WHERE CITY_NAME = "宜蘭縣";#排序(由小到大)select * from AIR_CONDITIONER_PROJECT WHERE CITY_NAME = "新北市" ORDER BY CLASSROOM_NUMBER ASC;#排序(由大到小)select * from AIR_CONDITIONER_PROJECT WHERE CITY_NAME = "新北市" ORDER BY CLASSROOM_NUMBER DESC;
更新資料(TABLE)#欄位都要用`才正確UPDATE `AIR_CONDITIONER_PROJECT`  SET `CITY_NAME` = "澎湖縣1" WHERE `CITY_NAME` = "澎湖縣";
